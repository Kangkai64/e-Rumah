import { createClient } from "npm:@supabase/supabase-js@2.28.0";
import process from "node:process";

const SUPABASE_URL = Deno.env.get('SUPABASE_URL') || process.env.SUPABASE_URL;
const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') || process.env.SUPABASE_SERVICE_ROLE_KEY;
if (!SUPABASE_URL || !SUPABASE_SERVICE_ROLE_KEY) {
  console.error('Missing SUPABASE_URL or SUPABASE_SERVICE_ROLE_KEY');
}

const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY, { auth: { persistSession: false } });

async function sendEmail(to: string, name: string, subject: string, html: string) {
  // Reuse 'enquiry-form-confirmation-email'; it requires {to, name, subject, message}
  // and 'emailType' picks the raw-HTML template (message is embedded as-is, no enquiry wording).
  const res = await supabase.functions.invoke('enquiry-form-confirmation-email', {
    body: { to, name, subject, message: html, emailType: 'health_report_share' },
  }).catch((e: any) => ({ error: e }));
  return res;
}

function escapeHtml(str: string) {
  return str
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;');
}

function formatDueDate(iso: string) {
  return new Intl.DateTimeFormat('en-MY', {
    weekday: 'long', year: 'numeric', month: 'long', day: 'numeric',
    hour: 'numeric', minute: '2-digit', hour12: true,
    timeZone: 'Asia/Kuala_Lumpur',
  }).format(new Date(iso));
}

// Lead time is derived from the gap between when the reminder fires and
// when it's due, rather than parsing notification_offset's interval text
// (Postgres/PostgREST interval serialization isn't guaranteed stable).
function formatLeadTime(scheduledIso: string, dueIso: string) {
  const diffDays = Math.round((new Date(dueIso).getTime() - new Date(scheduledIso).getTime()) / (1000 * 60 * 60 * 24));
  if (diffDays <= 0) return 'today';
  if (diffDays === 1) return 'tomorrow';
  if (diffDays === 7) return 'in 1 week';
  return `in ${diffDays} days`;
}

async function processNotifications() {
  const nowIso = new Date().toISOString();
  const { data, error } = await supabase
    .from('reminder_notifications')
    .select('id, reminder_id, scheduled_time, notification_offset, is_sent, reminder:reminders(user_id, reminder_title, reminder_date, reminder_type, category, notes)')
    .lte('scheduled_time', nowIso)
    .eq('is_sent', false)
    .limit(100);

  if (error) {
    console.error('Query error', error);
    return;
  }
  if (!data || (Array.isArray(data) && data.length === 0)) return;

  for (const row of data as any[]) {
    try {
      const userRes = await supabase.from('users').select('email, full_name').eq('id', row.reminder.user_id).single();
      if (userRes.error) throw userRes.error;
      const to = userRes.data.email;
      const name = userRes.data.full_name || 'there';

      const title = row.reminder.reminder_title;
      const type = row.reminder.reminder_type || 'Reminder';
      const category = row.reminder.category || 'Health & appointments';
      const dueText = formatDueDate(row.reminder.reminder_date);
      const leadTime = formatLeadTime(row.scheduled_time, row.reminder.reminder_date);

      const subject = `Reminder: ${title} is due ${leadTime}`;
      const notesBlock = row.reminder.notes
        ? `<p style="margin: 16px 0 0;"><strong>Notes:</strong><br>${escapeHtml(row.reminder.notes).replace(/\n/g, '<br>')}</p>`
        : '';
      const html = `
        <h2 style="color:#A8202D; margin-bottom:4px;">${escapeHtml(title)}</h2>
        <p style="color:#666; margin-top:0; font-size:13px;">${escapeHtml(type)} &middot; ${escapeHtml(category)}</p>
        <p>Hi ${escapeHtml(name)}, this is a reminder that your <strong>${escapeHtml(type.toLowerCase())}</strong> is due <strong>${leadTime}</strong>.</p>
        <div style="background-color:#f9f9f9; padding:16px; border-radius:8px; margin:16px 0;">
          <p style="margin:0;"><strong>Due:</strong> ${dueText}</p>
        </div>
        ${notesBlock}
      `;

      const sendRes = await sendEmail(to, name, subject, html);
      if (sendRes?.error) throw sendRes.error;

      const upd = await supabase
        .from('reminder_notifications')
        .update({ is_sent: true, sent_at: new Date().toISOString() })
        .eq('id', row.id);
      if (upd.error) throw upd.error;
    } catch (e) {
      console.error('Failed to send for notification', row.id, e);
    }
  }
}

console.info('reminder-processor starting');
Deno.serve(async (req: Request) => {
  const corsHeaders = () => ({
    'Access-Control-Allow-Origin': '*',
    'Access-Control-Allow-Methods': 'POST, GET, OPTIONS',
    'Access-Control-Allow-Headers': 'Content-Type, Authorization, apikey',
  });

  if (req.method === 'OPTIONS') return new Response(null, { status: 204, headers: corsHeaders() });

  try {
    const url = new URL(req.url);
    if (url.pathname.endsWith('/run')) {
      await processNotifications();
      return new Response(JSON.stringify({ status: 'ok' }), {
        headers: { ...corsHeaders(), 'Content-Type': 'application/json' },
      });
    }
    return new Response('reminder-processor: alive', { headers: corsHeaders() });
  } catch (err) {
    console.error('Handler error', err);
    return new Response('error', { status: 500, headers: corsHeaders() });
  }
});
