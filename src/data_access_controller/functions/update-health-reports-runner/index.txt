// Dependency-free Edge Function
// Calls the Supabase Functions REST endpoint for RPC using the service role key

const SUPABASE_URL = Deno.env.get('SUPABASE_URL')!;
const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!;

Deno.serve(async (req: Request) => {
  try {
    if (!SUPABASE_URL || !SUPABASE_SERVICE_ROLE_KEY) {
      return new Response(JSON.stringify({ status: 'error', error: 'Missing SUPABASE_URL or SUPABASE_SERVICE_ROLE_KEY' }), { headers: { 'Content-Type': 'application/json' }, status: 500 });
    }

    // Call Postgres RPC via the REST Admin endpoint
    const url = SUPABASE_URL.replace(/\/+$/,'') + '/rest/v1/rpc/update_health_reports_due_status';

    const resp = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'apikey': SUPABASE_SERVICE_ROLE_KEY,
        'Authorization': 'Bearer ' + SUPABASE_SERVICE_ROLE_KEY,
      },
      // RPC functions may accept an empty body
      body: JSON.stringify({}),
    });

    const text = await resp.text();
    let data;
    try { data = text ? JSON.parse(text) : null; } catch { data = text; }

    if (!resp.ok) {
      return new Response(JSON.stringify({ status: 'error', status_code: resp.status, body: data }), { headers: { 'Content-Type': 'application/json' }, status: 500 });
    }

    return new Response(JSON.stringify({ status: 'ok', status_code: resp.status, result: data }), { headers: { 'Content-Type': 'application/json' } });
  } catch (err) {
    console.error(err);
    return new Response(JSON.stringify({ status: 'error', error: String(err) }), { headers: { 'Content-Type': 'application/json' }, status: 500 });
  }
});
